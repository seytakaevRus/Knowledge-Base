const dominantIndex = (nums: number[]): number => {
    let maxValueIndex = 0;

    const maxValue = Math.max(...nums);

    const isSatisfy = nums.every((value, index) => {
        if (value === maxValue) {
            maxValueIndex = index;
            return true;
        }

        return value * 2 <= maxValue;
    });

    if (isSatisfy) return maxValueIndex;

    return -1;
};```
