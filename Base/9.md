### Answer 2

```typescript
const isPalindrome = (x: number): boolean => {
  if (x < 0) return false;
  if (x === 0) return true;

  let temp = x;
  let value = '';

  while (temp > 0) {
    value += temp % 10;

    temp = Math.trunc(temp / 10);
  }

  return String(x) === value;
};
```

### Answer 3

```typescript
const isPalindrome = (x: number): boolean => {
  if (x < 0) return false;

  let temp = x;
  let value = 0;

  while (temp > 0) {
    const lastDigit = temp % 10;

    value = value * 10 + lastDigit

    temp = Math.trunc(temp / 10);
  }

  return x === value;
};
```

Сложность по времени: `O()`.

Сложность по памяти: `O()`.
