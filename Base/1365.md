### Answer 1

```typescript
const smallerNumbersThanCurrent = (nums: number[]): number[] => {
	const output: number[] = [];

	for (let i = 0; i < nums.length; i += 1) {
		let smallerThanCount = 0;

		for (let j = 0; j < nums.length; j += 1) {
			if (i !== j && nums[j] < nums[i]) {
				smallerThanCount += 1;
			}
		}

		output[i] = smallerThanCount;
	}

	return output;
};
```
